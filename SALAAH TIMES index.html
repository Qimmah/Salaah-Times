            // Theme configurations

            // South African cities with coordinates

            const cities = {

                'Cape Town': { lat: -33.9249, lng: 18.4241 },

                'Johannesburg': { lat: -26.2041, lng: 28.0473 },

                'Durban': { lat: -29.8587, lng: 31.0218 },

                'Pretoria': { lat: -25.7479, lng: 28.2293 },

                'Port Elizabeth': { lat: -33.9608, lng: 25.6022 },

                'Bloemfontein': { lat: -29.0852, lng: 26.1596 }

            };

 

            // Calculate prayer times using simplified calculation

            const calculatePrayerTimes = (lat, lng, date) => {

                const times = {};

                const hour = 60 * 60 * 1000;

               

                const baseDate = new Date(date);

                baseDate.setHours(0, 0, 0, 0);

               

                times.fajr = new Date(baseDate.getTime() + 5 * hour + 30 * 60 * 1000);

                times.sunrise = new Date(baseDate.getTime() + 6 * hour + 45 * 60 * 1000);

                times.dhuhr = new Date(baseDate.getTime() + 12 * hour + 15 * 60 * 1000);

                times.asr = new Date(baseDate.getTime() + 15 * hour + 30 * 60 * 1000);

                times.maghrib = new Date(baseDate.getTime() + 18 * hour + 15 * 60 * 1000);

                times.isha = new Date(baseDate.getTime() + 19 * hour + 45 * 60 * 1000);

               

                return times;

            };

 

            // Format time for display

            const formatTime = (date) => {

                return date.toLocaleTimeString('en-US', {

                    hour: '2-digit',

                    minute: '2-digit',

                    hour12: false

                });

            };

 

            // Calculate next prayer

            const findNextPrayer = (times, current) => {

                const prayerNames = ['fajr', 'dhuhr', 'asr', 'maghrib', 'isha'];

                const prayerLabels = {

                    fajr: 'Fajr',

                    dhuhr: 'Dhuhr',

                    asr: 'Asr',

                    maghrib: 'Maghrib',

                    isha: 'Isha'

                };

 

                for (let prayer of prayerNames) {

                    if (times[prayer] && current < times[prayer]) {

                        const remaining = times[prayer] - current;

                        const hours = Math.floor(remaining / (1000 * 60 * 60));

                        const minutes = Math.floor((remaining % (1000 * 60 * 60)) / (1000 * 60));

                       

                        return {

                            name: prayerLabels[prayer],

                            time: formatTime(times[prayer]),

                            remaining: `${hours}h ${minutes}m`

                        };

                    }

                }

               

                const tomorrow = new Date(current);

                tomorrow.setDate(tomorrow.getDate() + 1);

                const tomorrowTimes = calculatePrayerTimes(cities[location].lat, cities[location].lng, tomorrow);

                const remaining = tomorrowTimes.fajr - current;

                const hours = Math.floor(remaining / (1000 * 60 * 60));

                const minutes = Math.floor((remaining % (1000 * 60 * 60)) / (1000 * 60));

               

                return {

                    name: 'Fajr',

                    time: formatTime(tomorrowTimes.fajr),

                    remaining: `${hours}h ${minutes}m`

                };

            };

 

            // Get Hijri date (simplified)

            const getHijriDate = () => {

                const gregorianDate = new Date();

                const hijriYear = gregorianDate.getFullYear() - 579;

                const months = ['Muharram', 'Safar', 'Rabi al-Awwal', 'Rabi al-Thani', 'Jumada al-Awwal',

                               'Jumada al-Thani', 'Rajab', 'Shaban', 'Ramadan', 'Shawwal', 'Dhu al-Qadah', 'Dhu al-Hijjah'];

                const month = months[gregorianDate.getMonth()];

                const day = gregorianDate.getDate();

               

                return `${day} ${month} ${hijriYear} AH`;

            };

 

            // Prayer icons

            const getPrayerIcon = (prayer) => {

                const iconProps = { size: 20, className: theme.iconColor };

                switch(prayer) {

                    case 'fajr': return React.createElement(Star, iconProps);

                    case 'sunrise': return React.createElement(Sunrise, iconProps);

                    case 'dhuhr': return React.createElement(Sun, iconProps);

                    case 'asr': return React.createElement(Sun, iconProps);

                    case 'maghrib': return React.createElement(Sunset, iconProps);

                    case 'isha': return React.createElement(Moon, iconProps);

                    default: return React.createElement(Clock, iconProps);

                }

            };

                }, 1000);

                const times = calculatePrayerTimes(cities[location].lat, cities[location].lng, currentTime);

                setHijriDate(getHijriDate());

            }, [location, currentTime]);

                                        const currentIndex = themeKeys.indexOf(currentTheme);

                                        const nextIndex = (currentIndex + 1) % themeKeys.length;

                                        setCurrentTheme(themeKeys[nextIndex]);
